@my-react

A React like framework, can be used to replace all the react/react-dom api for test、learn、debug ...

This website is built with @my-react project.
Version: @my-react/react [0.2.1]; @my-react/react-dom [0.2.1]

          tsx
          

        
1import { useState, useCallback } from '@my-react/react';
2import { render } from '@my-react/react-dom';
3
4const useCount = () => {
5  const [state, setState] = useState(0);
6  const add = useCallback(() => setState(i => i + 1), []);
7  const del = useCallback(() => setState(i => i - 1), []);
8
9  return [state, add, del];
10};
11
12const App = () => {
13  const [state, add, del] = useCount();
14
15  return <div>
16    <p>{state}</p>
17    <button onClick={add}>add</button>
18    <button onClick={del}>del</button>
19  </div>
20}
21
22render(<App />, document.querySelector('#root'));
